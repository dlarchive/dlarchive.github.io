1. DECLARE  
  2.     str VARCHAR(50);  
  3.     strA VARCHAR(50);  
  4.     strB VARCHAR(50);   
  5.     len NUMBER;   
  6. BEGIN  
  7.     -- A prompt window will ask you to enter your name  
  8.     -- In Oracle SQL Developer  
  9.     str := '&YOUR_NAME';  
  10.     strA := str;  
  11.     len := length(str);   
  12.   
  13.     -- Reversing the name  
  14.     FOR i IN REVERSE 1.. len  
  15.     LOOP                
  16.         strB := strB || substr(str, i, 1);   
  17.     END LOOP;   
  18.   
  19.     dbms_output.Put_line('The reverse of '|| strA ||' is '|| strB);   
  20. END;
