1. DECLARE      
  2.     TYPE yearList IS VARRAY(6) OF INTEGER;       
  3.     dateList yearList;      
  4.     i NUMBER(2);      
  5.           
  6. BEGIN      
  7.     i := 1;      
  8.     dateList := yearList(1990, 2000, 1996, 1900, 2016, 1884);       
  9.           
  10.     LOOP      
  11.           
  12.         IF (((dateList(i) MOD 4 = 0) AND (dateList(i) MOD 100 != 0))    
  13.         OR (dateList(i) MOD 400 = 0)) THEN      
  14.            dbms_output.put_line(dateList(i) || ' is leap year');      
  15.         ELSE      
  16.            dbms_output.put_line(dateList(i) || ' is not leap year');      
  17.         END IF;      
  18.       
  19.         i := i+1;    
  20.         EXIT WHEN i > 6;      
  21.   
  22.     END LOOP;  
  23.           
  24. END;
