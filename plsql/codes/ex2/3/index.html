DECLARE TYPE yearList IS VARRAY(6) OF INTEGER; dateList yearList; i NUMBER(2); BEGIN i := 1; dateList := yearList(1990, 2000, 1996, 1900, 2016, 1884); LOOP IF (((dateList(i) MOD 4 = 0) AND (dateList(i) MOD 100 != 0)) OR (dateList(i) MOD 400 = 0)) THEN dbms_output.put_line(dateList(i) || ' is leap year'); ELSE dbms_output.put_line(dateList(i) || ' is not leap year'); END IF; i := i+1; EXIT WHEN i > 6; END LOOP; END;
